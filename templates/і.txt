    <script>
        // Завантаження операцій
        fetch('/api/operations')
            .then(response => response.json())
            .then(data => {
                const tableBody = document.getElementById('operation-table');
                data.forEach(op => {
                    const row = `
                        <tr>
                            <td>${op.date}</td>
                            <td>${op.time}</td>
                            <td>${op.operating_room}</td>
                            <td>${op.operation_name}</td>
                            <td>${op.priority}</td>
                            <td>${op.surgeon}</td>
                            <td>${op.anesthesiologist}</td>
                            <td>${op.first_assistant}, ${op.second_assistant}</td>
                        </tr>`;
                    tableBody.innerHTML += row;
                });
            });
    </script>
